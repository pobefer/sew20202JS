            public static int busquedaBinaria(int a[], int toSearch) {
                //O(log(n))
               int low = 0;
               int high = a.length - 1;
               int medio = 0;
               while(high > low) {
                   medio = (low + high) / 2;
                   if( a[medio] == toSearch )
                       return medio;
                   else if( a[medio] > toSearch )
                       high = medio - 1;
                   else
                       low = medio + 1;
               }
               return -1;
            }
